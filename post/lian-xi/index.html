（2011年预赛题) 由键盘输入一个自然数Ｎ（1≤N≤9），要求程序在屏幕上输出如下图形。

Ｎ＝１时：　　　　　　　　Ｎ＝３时：　　　　　　　Ｎ＝４时：
　　１　　　　　　　　　　　　１１１１１　　　　　　　１１１１１１１
　　　　　　　　　　　　　　　１２２２１　　　　　　　１２２２２２１
　　　　　　　　　　　　　　　１２３２１　　　　　　　１２３３３２１
　　　　　　　　　　　　　　　１２２２１　　　　　　　１２３４３２１
　　　　　　　　　　　　　　　１１１１１　　　　　　　１２３３３２１
　　　　　　　　　　　　　　　　　　　　　　　　　　　１２２２２２１
　　　　　　　　　　　　　　　　　　　　　　　　　　　１１１１１１１
　　注：应保证Ｎ在图形中只出现一次，且在中心。

#include <iostream>
using namespace std;

int main()
{
	int n = 0;
	cout << "输入【1,9】的数： ";
	int a[20][20];
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		for(int row=i;row<=2*n-2-i;row++)
			for (int col = i; col <= 2 * n - 2 - i; col++)
			{
				a[row][col] = i + 1;
			}
	}
	
	// 输出
	for (int i = 0; i < 2*n-1; i++)
	{
		for (int j = 0; j < 2*n-1; j++)
			cout << a[i][j] << " ";
		cout << endl;
	}
}
